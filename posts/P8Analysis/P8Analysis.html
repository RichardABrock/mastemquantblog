Grouped Scatter plot with marginal density plots

library(ggpubr)

# Grouped Scatter plot with marginal density plots

plotdata<-dfe2023%>%
  filter(ICLOSE==0)%>%
  filter(P8MEA!="NP")%>%
  filter(P8MEA!="NE")%>%
  filter(P8MEA!="SUPP")%>%
  select(P8MEA, PTFSM6CLA1A, ADMPOL,SCHNAME)%>%
  na.omit()%>%
  mutate(label=ifelse(P8MEA>1.3,SCHNAME,""))%>%
  mutate(ADMPOL=ifelse(ADMPOL=="NR","Not recorded",
                       ifelse(ADMPOL=="SEL","Selective", 
                              ifelse(ADMPOL=="NSE","Not selective", NA))))
plotdata$P8MEA<-as.numeric(plotdata$P8MEA)

plotdata$PTFSM6CLA1A<-parse_number(plotdata$PTFSM6CLA1A)

ggscatterhist(data=plotdata, y="P8MEA", x="PTFSM6CLA1A",
              color ="ADMPOL", size = 0.5, alpha = 0.4,
              margin.params = list(fill = "ADMPOL", color = "black", size = 0.2),
              ggtheme = theme_bw())
